# git
cd ..                              返回上一层
pwd                                查看当前目录
ls                                 列出当前目录下所有文件
touch                              创建文件
rm                                 删除文件
mkdir                              创建文件夹
rm -r                              删除文件夹E
mv                                 移动文件
reset                              重置 初始化
history                            查看历史命令
git branch                         查看当前所在分支
git remote show origin             查看所有远程库
